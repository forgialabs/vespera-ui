UI is inside a `.vsp-root`. Try them in the [live playground](/demo).

## Settings row with a toggle

```tsx

function TwoFactorRow() {
  const [on, setOn] = useState(true);
  return (
    <div className="ui-setrow">
      <div className="ui-setrow-main">
        <div className="ui-setrow-title">Two-factor auth</div>
        <div className="ui-setrow-desc">Require a code at sign-in.</div>
      </div>
      <Switch checked={on} onChange={setOn} aria-label="Two-factor auth" />
    </div>
  );
}
```

## Confirm dialog with a toast

```tsx

function DeleteButton() {
  const [open, setOpen] = useState(false);
  return (
    <>
      <Button variant="danger" onClick={() => setOpen(true)}>Delete</Button>
      <Dialog
        open={open}
        onClose={() => setOpen(false)}
        tone="neg"
        icon={<Icon.shield />}
        title="Delete workspace?"
        desc="This cannot be undone."
        footer={
          <>
            <Button onClick={() => setOpen(false)}>Cancel</Button>
            <Button
              variant="danger"
              onClick={() => {
                setOpen(false);
                toast({ tone: 'neg', title: 'Deleted' });
              }}
            >
              Delete
            </Button>
          </>
        }
      />
      <ToastHost />
    </>
  );
}
```

## Command palette (⌘K)

```tsx

function App() {
  const [open, setOpen] = useState(false);
  useCmdK(setOpen); // ⌘K / Ctrl+K toggles it
  return (
    <CommandPalette
      open={open}
      onClose={() => setOpen(false)}
      groups={[
        {
          label: 'Actions',
          items: [
            { label: 'New project', icon: <Icon.plus />, meta: '⌘N', onRun: () => {} },
            { label: 'Search docs', icon: <Icon.doc />, keywords: 'help' },
          ],
        },
      ]}
    />
  );
}
```
