import { Card, CardGrid } from '@astrojs/starlight/components';
import manifest from '@manifest/react.json';
import { slugOf, CATEGORY_BLURB } from '../../../lib/categories.ts';

export const DEMO_CATS = new Set([
  'Actions',
  'Forms & inputs',
  'Date & time',
  'Data display',
  'Charts',
  'Navigation',
  'Feedback & status',
  'Overlays',
  'Layout & structure',
]);
export const groups = manifest.categories.map((label) => ({
  label,
  blurb: CATEGORY_BLURB[label],
  count: manifest.components.filter((c) => c.category === label).length,
  href: DEMO_CATS.has(label) ? `/vespera-ui/components/${slugOf(label)}/` : '/vespera-ui/examples/',
}));

Live examples you can interact with, grouped by category — the same grouping as the
[API reference](/vespera-ui/reference/react/). Everything renders inside a themed `.vsp-root`; see
[Installation](/vespera-ui/guides/installation/) to set up and
[Foundations](/vespera-ui/foundations/color/) for the visual system behind it.

<CardGrid>
  {groups.map((g) => (
    <Card title={`${g.label} · ${g.count}`}>
      {g.blurb}
      <p><a href={g.href}>{g.label === 'Blocks' ? 'See examples →' : 'View examples →'}</a></p>
    </Card>
  ))}
</CardGrid>

Composed sections (Orders, Kanban, API keys…) live on the
[Examples & templates](/vespera-ui/examples/) page, and the full prop tables are in the
[API reference](/vespera-ui/reference/react/).
