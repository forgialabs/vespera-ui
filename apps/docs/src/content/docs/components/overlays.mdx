Every overlay portals into the nearest `.vsp-root` and is `position: fixed`. Each demo runs in an
isolated frame so whatever it pops open stays inside the preview.

## Dialog, menu & toast

<Preview
  code={`import { Button, Dialog, DropdownMenu, ToastHost, toast } from '@vespera-ui/react';
import { Icon } from '@vespera-ui/icons';

<Button variant="primary" onClick={() => setOpen(true)}>Open dialog</Button>
<DropdownMenu trigger={<Button trailingIcon={<Icon.chevDown />}>Menu</Button>} items={[
  { label: 'Profile', icon: <Icon.user /> },
  { sep: true },
  { label: 'Log out', icon: <Icon.logout />, danger: true },
]} />
<Button onClick={() => toast({ tone: 'pos', title: 'Saved' })}>Fire a toast</Button>
<Dialog open={open} onClose={() => setOpen(false)} tone="neg" title="Delete workspace?" /* … */ />
<ToastHost />`}
  demo="overlay"
/>

## Command palette

<Preview
  code={`import { CommandPalette, useCmdK } from '@vespera-ui/react';

const [open, setOpen] = useState(false);
useCmdK(setOpen); // ⌘K / Ctrl+K

<CommandPalette open={open} onClose={() => setOpen(false)} groups={[
  { label: 'Actions', items: [
    { label: 'New project', icon: <Icon.plus />, meta: '⌘N' },
    { label: 'Search docs', icon: <Icon.doc />, keywords: 'help' },
  ]},
]} />`}
  demo="command"
/>
